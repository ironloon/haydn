```rust
const NOTE_NAMES: [&str; 12] = ["C", "C#", "D", "Eb", "E", "F", "F#", "G", "Ab", "A", "Bb", "B"];
fn note_name(midi: u8) -> String {
    let name = NOTE_NAMES[(midi % 12) as usize];
    let octave = (midi as i16 / 12) - 1;
    format!("{}{}", name, octave)
}
```

3. **Define MIDI callback as a named function** (not a closure — enables reuse for reconnect in Plan 02 per D-10):
```rust
fn midi_callback(_timestamp: u64, message: &[u8], tx: &mut std::sync::mpsc::Sender<MidiMsg>) {
    if let Ok(event) = midly::live::LiveEvent::parse(message) {
        if let midly::live::LiveEvent::Midi { message: msg, .. } = event {
            match msg {
                midly::MidiMessage::NoteOn { key, vel } => {
                    if vel.as_int() == 0 {
                        // NoteOn with velocity 0 = NoteOff (MIDI convention, per D-16)
                        let _ = tx.send(MidiMsg::NoteOff { note: key.as_int() });
                    } else {
                        let _ = tx.send(MidiMsg::NoteOn {
                            note: key.as_int(),
                            velocity: vel.as_int(),
                        });
                    }
                }
                midly::MidiMessage::NoteOff { key, .. } => {
                    let _ = tx.send(MidiMsg::NoteOff { note: key.as_int() });
                }
                _ => {} // Ignore CC, pitch bend, etc. (deferred per CONTEXT)
            }
        }
    }
}
```
The `let _ =` on send silently drops if receiver is gone, avoiding callback panics.

7. **Implement main event loop** (per D-01, D-03):
```rust
let mut vm = haydn_vm::HaydnVm::new();
println!("Listening for MIDI input... (Ctrl+C to quit)\n");

while running.load(std::sync::atomic::Ordering::Relaxed) {
    match rx.recv_timeout(std::time::Duration::from_millis(500)) {
        Ok(MidiMsg::NoteOn { note, velocity }) => {
            // Per D-15: velocity logged but not passed to map_note
            match engine.map_note(note) {
                Some(event) => {
                    let results = vm.process_event(event);
                    for result in &results {
                        print_event_log(note, velocity, result);
                    }
                }
                None => {
                    println!("[{} v={}] → (unmapped)", note_name(note), velocity);
                }
            }
        }
        Ok(MidiMsg::NoteOff { note }) => {
            // Per D-16: logged but discarded, no VM event
            println!("[{} released]", note_name(note));
        }
        Err(std::sync::mpsc::RecvTimeoutError::Timeout) => continue,
        Err(std::sync::mpsc::RecvTimeoutError::Disconnected) => break,
    }
}

println!("\nSession ended. Final stack: {:?}", vm.stack());
```

```rust
fn print_event_log(note: u8, velocity: u8, result: &haydn_vm::StepResult)
```
- Base format: `[{note_name} v={velocity}] → {mapping}  |  Stack: {stack}`

- If `result.edge_case` is `Some(ec)`: append `  |  ⚠ {ec:?}` (e.g., `⚠ StackUnderflow`)
- For loop-related operations (LoopEntered, LoopExited, LoopReplaying, etc.): show the operation type in the mapping field instead of the raw event
